4   import com.nnk.springboot.services.CurvePointService;
5   import org.springframework.beans.factory.annotation.Autowired;
6   import org.springframework.stereotype.Controller;
7   import org.springframework.ui.Model;
8   import org.springframework.validation.BindingResult;
9   import org.springframework.web.bind.annotation.*;
10  import org.springframework.web.servlet.mvc.support.RedirectAttributes;
11  
12  import javax.validation.Valid;
13  import java.util.List;
14  
15  @Controller
16  public class CurveController {
17  
18      @Autowired
19      private CurvePointService curvePointService;
20  
21      public CurveController(CurvePointService curvePointService) {
22          this.curvePointService = curvePointService;
23      }
24  
25      @RequestMapping("/curvePoint/list")
26      public String home(Model model)
27      {
28          List<CurvePoint> curvePointList = curvePointService.findAll();
29          model.addAttribute("curvePoints",curvePointList);
30          return "curvePoint/list";
31      }
32  
33      @GetMapping("/curvePoint/add")
34      public String addCurveForm(CurvePoint bid,@RequestParam(value = "errorcurve",defaultValue = "")String errorcurve,Model model) {
35          model.addAttribute("errorcurve", errorcurve);
36          return "curvePoint/add";
37      }
38  
39      @PostMapping("/curvePoint/validate")
40      public String validate(@Valid CurvePoint curvePoint,@RequestParam(value="action",required = true)String action, BindingResult result, Model model, RedirectAttributes redirectAttributes) {
41          String redirect = "";
42          if (curvePoint.getCurveId()<=0){
43              redirectAttributes.addAttribute("errorcurve","must not be null");
44              return "redirect:/curvePoint/add";
45          }else {
46              if (action.equals("cancel")){
47                 curvePoint.setCurveId(0);
48                 curvePoint.setTerm(0);
49                 curvePoint.setValue(0);
50                  return  "curvePoint/add";
51              }
52              if (action.equals("add")){
53                  curvePointService.save(curvePoint);
54                  redirect ="redirect:/curvePoint/list";
55              }
56          }
57          return redirect;
58      }
59  
60      @GetMapping("/curvePoint/update/{id}")
61      public String showUpdateForm(@PathVariable("id") Integer id,@RequestParam(value = "errorcurve",defaultValue = "")String errorcurve, Model model) {
62          model.addAttribute("errorcurve",errorcurve);
63          CurvePoint curvePoint = curvePointService.findById(id);
64          model.addAttribute("curvePoint",curvePoint);
65          return "curvePoint/update";
66      }
67  
68      @PostMapping("/curvePoint/update/{id}")
69      public String updateCurve(@PathVariable("id") Integer id, @Valid CurvePoint curvePoint,@RequestParam(value="action",required = true)String action,
70                               BindingResult result, Model model, RedirectAttributes redirectAttributes) {
71          String redirect = "";
72          if (curvePoint.getCurveId()<=0){
73              redirectAttributes.addAttribute("errorcurve","must not be null");
74              return "redirect:/curvePoint/update/{id}";
75          }else {
76              if (action.equals("cancel")){
77                  return  "redirect:/curvePoint/update/"+id;
78              }
79              if (action.equals("update")){
80                  curvePointService.update(curvePoint,id);
81                  redirect ="redirect:/curvePoint/list";
82              }
83          }
84          return redirect;
85      }
86  
87      @GetMapping("/curvePoint/delete/{id}")
88      public String deleteCurve(@PathVariable("id") Integer id, Model model) {
89          curvePointService.delete(id);
90          return "redirect:/curvePoint/list";
91      }
92  }
