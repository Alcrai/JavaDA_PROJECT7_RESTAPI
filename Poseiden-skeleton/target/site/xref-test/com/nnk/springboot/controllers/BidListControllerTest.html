49      }

55      public void validateTest(){
56          BidList bidNull= new BidList("","",10d);
57          assertThat(bidListController.validate(bidNull,"add",bindingResult,model,redirectAttributes)).isEqualTo("redirect:/bidList/add");
58          assertThat(bidListController.validate(bid,"cancel",bindingResult,model,redirectAttributes)).isEqualTo("bidList/add");
59      }
60      @Test
61      public void showUpdateFormTest(){
62          when(bidListService.findById(1)).thenReturn(bid);
63          assertThat(bidListController.showUpdateForm(1,"","",model)).isEqualTo("bidList/update");
64          verify(bidListService).findById(1);
65      }
66      @Test
67      public void updateBidTest(){

69          assertThat(bidListController.updateBid(1,bidNull,"add",bindingResult,model,redirectAttributes)).isEqualTo("redirect:/bidList/update/{id}");
70          assertThat(bidListController.updateBid(1,bid,"cancel",bindingResult,model,redirectAttributes)).isEqualTo("redirect:/bidList/update/1");
71      }
72      @Test
73      public void deleteBidtest(){
